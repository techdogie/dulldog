     print("Obtained marks = ", marks);

     per = (marks / tmarks) * 100;

     print("Percentage of Student is ", per);

     if(per >= 80):

     elif(per >= 70 or per == 60):
        print(" Grade B");

     elif(per >= 40):
        print(" Grade D");

            void ddaline(int x1, int y1, int x2, int y2)
            {
               float dx, dy, steps, xinc, yinc;
               float x=x1, y=y1, i;
               dx=abs(x1-x2);
               dy=abs(y1-y2);
               if(dx>dy)
               {
                 steps=dx;
               }
               else
               {
                 steps=dy;
               }
               xinc=dx/steps;
               yinc=dy/steps;
               putpixel(x,y,RED);
               for(i=1;i<=steps;i++)
               {
                 x=x+xinc;
                 y=y+yinc;
                 putpixel(x,y,RED);
                 // delay(100);
               }
            }
